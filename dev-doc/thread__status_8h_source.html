40 // A structure that describes the current status of a thread.
41 // The status of active threads can be fetched using
42 // ROCKSDB_NAMESPACE::GetThreadList().
43 struct ThreadStatus {
44  // The type of a thread.
45  enum ThreadType : int {
46  HIGH_PRIORITY = 0, // RocksDB BG thread in high-pri thread pool
47  LOW_PRIORITY, // RocksDB BG thread in low-pri thread pool
48  USER, // User thread (Non-RocksDB BG thread)
49  BOTTOM_PRIORITY, // RocksDB BG thread in bottom-pri thread pool
50  NUM_THREAD_TYPES
51  };
52 
53  // The type used to refer to a thread operation.
54  // A thread operation describes high-level action of a thread.
55  // Examples include compaction and flush.
56  enum OperationType : int {
57  OP_UNKNOWN = 0,
58  OP_COMPACTION,
59  OP_FLUSH,
60  NUM_OP_TYPES
61  };
62 
63  enum OperationStage : int {
64  STAGE_UNKNOWN = 0,
65  STAGE_FLUSH_RUN,
66  STAGE_FLUSH_WRITE_L0,
67  STAGE_COMPACTION_PREPARE,
68  STAGE_COMPACTION_RUN,
69  STAGE_COMPACTION_PROCESS_KV,
70  STAGE_COMPACTION_INSTALL,
71  STAGE_COMPACTION_SYNC_FILE,
72  STAGE_PICK_MEMTABLES_TO_FLUSH,
73  STAGE_MEMTABLE_ROLLBACK,
74  STAGE_MEMTABLE_INSTALL_FLUSH_RESULTS,
75  NUM_OP_STAGES
76  };
77 
78  enum CompactionPropertyType : int {
79  COMPACTION_JOB_ID = 0,
80  COMPACTION_INPUT_OUTPUT_LEVEL,
81  COMPACTION_PROP_FLAGS,
82  COMPACTION_TOTAL_INPUT_BYTES,
83  COMPACTION_BYTES_READ,
84  COMPACTION_BYTES_WRITTEN,
85  NUM_COMPACTION_PROPERTIES
86  };
87 
88  enum FlushPropertyType : int {
89  FLUSH_JOB_ID = 0,
90  FLUSH_BYTES_MEMTABLES,
91  FLUSH_BYTES_WRITTEN,
92  NUM_FLUSH_PROPERTIES
93  };
94 
95  // The maximum number of properties of an operation.
96  // This number should be set to the biggest NUM_XXX_PROPERTIES.
97  static const int kNumOperationProperties =
98  constexpr_max<NUM_COMPACTION_PROPERTIES, NUM_FLUSH_PROPERTIES>::result;
99 
100  // The type used to refer to a thread state.
101  // A state describes lower-level action of a thread
102  // such as reading / writing a file or waiting for a mutex.
103  enum StateType : int {
104  STATE_UNKNOWN = 0,
105  STATE_MUTEX_WAIT = 1,
106  NUM_STATE_TYPES
107  };
108 
109  ThreadStatus(const uint64_t _id, const ThreadType _thread_type,
110  const std::string& _db_name, const std::string& _cf_name,
111  const OperationType _operation_type,
112  const uint64_t _op_elapsed_micros,
113  const OperationStage _operation_stage,
114  const uint64_t _op_props[], const StateType _state_type)
115  : thread_id(_id),
116  thread_type(_thread_type),
117  db_name(_db_name),
118  cf_name(_cf_name),
119  operation_type(_operation_type),
120  op_elapsed_micros(_op_elapsed_micros),
121  operation_stage(_operation_stage),
122  state_type(_state_type) {
123  for (int i = 0; i < kNumOperationProperties; ++i) {
124  op_properties[i] = _op_props[i];
125  }
126  }
127 
128  // An unique ID for the thread.
129  const uint64_t thread_id;
130 
131  // The type of the thread, it could be HIGH_PRIORITY,
132  // LOW_PRIORITY, and USER
133  const ThreadType thread_type;
134 
135  // The name of the DB instance where the thread is currently
136  // involved with. It would be set to empty string if the thread
137  // does not involve in any DB operation.
138  const std::string db_name;
139 
140  // The name of the column family where the thread is currently
141  // It would be set to empty string if the thread does not involve
142  // in any column family.
143  const std::string cf_name;
144 
145  // The operation (high-level action) that the current thread is involved.
146  const OperationType operation_type;
147 
148  // The elapsed time of the current thread operation in microseconds.
149  const uint64_t op_elapsed_micros;
150 
151  // An integer showing the current stage where the thread is involved
152  // in the current operation.
153  const OperationStage operation_stage;
154 
155  // A list of properties that describe some details about the current
156  // operation. Same field in op_properties[] might have different
157  // meanings for different operations.
158  uint64_t op_properties[kNumOperationProperties];
159 
160  // The state (lower-level action) that the current thread is involved.
161  const StateType state_type;
162 
163  // The followings are a set of utility functions for interpreting
164  // the information of ThreadStatus
165 
166  static std::string GetThreadTypeName(ThreadType thread_type);
167 
168  // Obtain the name of an operation given its type.
169  static const std::string& GetOperationName(OperationType op_type);
170 
171  static const std::string MicrosToString(uint64_t op_elapsed_time);
172 
173  // Obtain a human-readable string describing the specified operation stage.
174  static const std::string& GetOperationStageName(OperationStage stage);
175 
176  // Obtain the name of the "i"th operation property of the
177  // specified operation.
178  static const std::string& GetOperationPropertyName(OperationType op_type,
179  int i);
180 
181  // Translate the "i"th property of the specified operation given
182  // a property value.
183  static std::map<std::string, uint64_t> InterpretOperationProperties(
184  OperationType op_type, const uint64_t* op_properties);
185 
186  // Obtain the name of a state given its type.
187  static const std::string& GetStateName(StateType state_type);
188 };
