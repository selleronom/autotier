19 // WALFilter allows an application to inspect write-ahead-log (WAL)
20 // records or modify their processing on recovery.
21 // Please see the details below.
22 class WalFilter : public Customizable {
23  public:
24  static const char* Type() { return "WalFilter"; }
25  static Status CreateFromString(const ConfigOptions& options,
26  const std::string& value, WalFilter** result);
27  enum class WalProcessingOption {
28  // Continue processing as usual
29  kContinueProcessing = 0,
30  // Ignore the current record but continue processing of log(s)
31  kIgnoreCurrentRecord = 1,
32  // Stop replay of logs and discard logs
33  // Logs won't be replayed on subsequent recovery
34  kStopReplay = 2,
35  // Corrupted record detected by filter
36  kCorruptedRecord = 3,
37  // Marker for enum count
38  kWalProcessingOptionMax = 4
39  };
40 
41  virtual ~WalFilter() {}
42 
43  // Provide ColumnFamily->LogNumber map to filter
44  // so that filter can determine whether a log number applies to a given
45  // column family (i.e. that log hasn't been flushed to SST already for the
46  // column family).
47  // We also pass in name->id map as only name is known during
48  // recovery (as handles are opened post-recovery).
49  // while write batch callbacks happen in terms of column family id.
50  //
51  // @params cf_lognumber_map column_family_id to lognumber map
52  // @params cf_name_id_map column_family_name to column_family_id map
53 
54  virtual void ColumnFamilyLogNumberMap(
55  const std::map<uint32_t, uint64_t>& /*cf_lognumber_map*/,
56  const std::map<std::string, uint32_t>& /*cf_name_id_map*/) {}
57 
58  // LogRecord is invoked for each log record encountered for all the logs
59  // during replay on logs on recovery. This method can be used to:
60  // * inspect the record (using the batch parameter)
61  // * ignoring current record
62  // (by returning WalProcessingOption::kIgnoreCurrentRecord)
63  // * reporting corrupted record
64  // (by returning WalProcessingOption::kCorruptedRecord)
65  // * stop log replay
66  // (by returning kStop replay) - please note that this implies
67  // discarding the logs from current record onwards.
68  //
69  // @params log_number log_number of the current log.
70  // Filter might use this to determine if the log
71  // record is applicable to a certain column family.
72  // @params log_file_name log file name - only for informational purposes
73  // @params batch batch encountered in the log during recovery
74  // @params new_batch new_batch to populate if filter wants to change
75  // the batch (for example to filter some records out,
76  // or alter some records).
77  // Please note that the new batch MUST NOT contain
78  // more records than original, else recovery would
79  // be failed.
80  // @params batch_changed Whether batch was changed by the filter.
81  // It must be set to true if new_batch was populated,
82  // else new_batch has no effect.
83  // @returns Processing option for the current record.
84  // Please see WalProcessingOption enum above for
85  // details.
86  virtual WalProcessingOption LogRecordFound(
87  unsigned long long /*log_number*/, const std::string& /*log_file_name*/,
88  const WriteBatch& batch, WriteBatch* new_batch, bool* batch_changed) {
89  // Default implementation falls back to older function for compatibility
90  return LogRecord(batch, new_batch, batch_changed);
91  }
92 
93  // Please see the comments for LogRecord above. This function is for
94  // compatibility only and contains a subset of parameters.
95  // New code should use the function above.
96  virtual WalProcessingOption LogRecord(const WriteBatch& /*batch*/,
97  WriteBatch* /*new_batch*/,
98  bool* /*batch_changed*/) const {
99  return WalProcessingOption::kContinueProcessing;
100  }
101 
102  // Returns a name that identifies this WAL filter.
103  // The name will be printed to LOG file on start up for diagnosis.
104  virtual const char* Name() const = 0;
105 };
